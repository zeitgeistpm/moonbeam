Helper to write EvmData impl for Solidity structs. Identifiers used should match Solidity ones. Types are infered from context, which should always be possible when parsing input to build a Rust struct.

ⓘ
impl EvmData for Currency {
	fn read(reader: &mut EvmDataReader) -> MayRevert<Self> {
		read_struct!(reader, (address, amount));
		Ok(Currency { address, amount })
	}

	fn write(writer: &mut EvmDataWriter, value: Self) {
		EvmData::write(writer, (value.address, value.amount));
	}

	fn has_static_size() -> bool {
		<(Address, U256)>::has_static_size()
	}
}
